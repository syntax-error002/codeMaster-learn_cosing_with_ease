15. Implement a Virtual Scrolling Component

class VirtualScroller {
constructor(container, items, itemHeight) {
this.container = container;
this.items = items;
this.itemHeight = itemHeight;
this.visibleItems = Math.ceil(container.clientHeight / itemHeight);
this.totalHeight = items.length * itemHeight;
this.scrollTop = 0;
this.init();
}
init() {
this.container.style.height = `${this.totalHeight}px`;
this.container.addEventListener('scroll', this.onScroll.bind(this));
this.render();
}
onScroll(event) {
this.scrollTop = event.target.scrollTop;
this.render();
}
render() {
const startIndex = Math.floor(this.scrollTop / this.itemHeight);
const endIndex = Math.min(
startIndex + this.visibleItems + 1,
this.items.length
);
// Render only visible items
const visibleContent = this.items
.slice(startIndex, endIndex)
.map(item => '
' +
item +
'
')
.join('');
this.container.innerHTML = visibleContent;
}
}

Performance Optimizations:

  • DOM Recycling
  • RAF for Smooth Scrolling
  • Debouncing Scroll Events
  • Height Calculations Caching

16. Implement a Service Worker for Offline Functionality

// service-worker.js
const CACHE_NAME = 'v1';
const URLS_TO_CACHE = [
'/', '/styles/main.css', '/scripts/app.js', '/images/logo.png'
];
self.addEventListener('install', event => {
event.waitUntil( caches.open(CACHE_NAME) .then(cache => cache.addAll(URLS_TO_CACHE)) );
});
self.addEventListener('fetch', event => {
event.respondWith( caches.match(event.request)
.then(response => {
if (response) {
return response;
} return fetch(event.request)
.then(response => {
if(!response || response.status !== 200) {
return response;
} const responseToCache = response.clone(); caches.open(CACHE_NAME)
.then(cache => {
cache.put(event.request, responseToCache);
});
return response; }); }) );
});
self.addEventListener('activate', event => {
event.waitUntil(
caches.keys().then(cacheNames => {
return Promise.all(
cacheNames.map(cacheName => {
if (cacheName !== CACHE_NAME) {
return caches.delete(cacheName);
}
})
);
})
);
});

Key Features:

  • Cache Management
  • Offline First Strategy
  • Background Sync
  • Push Notifications
Implementation Strategies:
  • Cache First
  • Network First
  • Stale While Revalidate
  • Cache Only
  • Network Only
